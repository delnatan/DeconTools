np.random.seed(42)
x = np.arange(512)


def make_filtered_background(x, correlation_length=20.0):
    # Start with random walk
    rw = np.cumsum(np.random.normal(0, 1, len(x)))

    # Create Gaussian filter
    x_filter = np.linspace(
        -5 * correlation_length,
        5 * correlation_length,
        int(10 * correlation_length / np.diff(x)[0]),
    )
    smooth_kernel = np.exp(-(x_filter**2) / (2 * correlation_length**2))
    smooth_kernel = smooth_kernel / smooth_kernel.sum()

    # Filter and add offset to keep positive
    background = np.convolve(rw, smooth_kernel, mode="same")
    background = background - background.min() + 100
    return background


# Create Lorentzian spikes
def lorentzian(x, x0, gamma, amplitude):
    return amplitude * gamma**2 / ((x - x0) ** 2 + gamma**2)


bg = make_filtered_background(x)

signal = sum(
    [
        lorentzian(x, 125, 5.0, 500.0),
        lorentzian(x, 200, 3.5, 720.0),
        lorentzian(x, 300, 2.5, 400.0),
        lorentzian(x, 320, 3.0, 300.0),
    ]
)

# create psf
ν = fft.fftshift(fft.fftfreq(x.size) * x.size)
psf = np.exp(-(ν**2) / (2 * 8.0**2))
psf /= psf.sum()

clean_signal = signal + bg
blurred_signal = np.convolve(signal, psf, mode="same") + bg
quantized_signal = np.random.poisson(lam=blurred_signal)
noisy_data = quantized_signal + bg + np.random.randn(x.size) * 1.25
